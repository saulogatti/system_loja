Future<void> loadUsuarios() async {
  final result = await _userRepository.obterTodosUsuarios();
  result.when(
    onSuccess: (usuarios) {
      emit(UsuarioState.loadSuccess(usuarios: usuarios));
    },
    onError: (message) {
      emit(
        UsuarioState.loadFailure(errorMessage: message),
      );
    },
  );
}
