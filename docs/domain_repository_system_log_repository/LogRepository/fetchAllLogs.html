@override
Future<ResultStatus<List<ActivityLog>, String>> fetchAllLogs() async {
  try {
    final result = await _logDao.getAll();
    return ResultStatus.success(result);
  } catch (e) {
    return ResultStatus.error('Falha ao obter logs: $e');
  }
}
