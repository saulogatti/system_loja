Future<List<InvoiceItem>> getByInvoiceId(int invoiceId) async {
  final records = await (select(
    invoiceItemsRecords,
  )..where((t) => t.invoiceId.equals(invoiceId))).get();
  return records.map((record) => record.toDomain()).toList();
}
